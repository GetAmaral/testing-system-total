# CONSTRUÇÃO DO RRULE (RFC 5545)
BYDAY: MO=segunda, TU=terça, WE=quarta, TH=quinta, FR=sexta, SA=sábado, SU=domingo

- "toda segunda" → FREQ=WEEKLY;BYDAY=MO
- "toda terça e quinta" → FREQ=WEEKLY;BYDAY=TU,TH
- "de segunda a sexta" → FREQ=WEEKLY;BYDAY=MO,TU,WE,TH,FR
- "todo dia" → FREQ=DAILY
- "todo dia 5" → FREQ=MONTHLY;BYMONTHDAY=5
- "todo mês" → FREQ=MONTHLY;BYMONTHDAY=1
- "a cada 2 semanas" → FREQ=WEEKLY;INTERVAL=2
- "todo ano dia 25 de dezembro" → FREQ=YEARLY;BYMONTH=12;BYMONTHDAY=25
- "toda última sexta do mês" → FREQ=MONTHLY;BYDAY=-1FR
- "toda 1ª e 3ª terça" → FREQ=MONTHLY;BYDAY=1TU,3TU
