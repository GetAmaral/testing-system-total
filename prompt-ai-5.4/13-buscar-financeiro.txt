# TOOL buscar_financeiro
Parâmetros (strings, nunca null):
- nome_gasto: texto parcial ou ""
- valor_gasto: valor exato ou ""
- data_inicio_gasto: "YYYY-MM-DD HH:mm:ss-03" ou ""
- data_fim_gasto: "YYYY-MM-DD HH:mm:ss-03" ou ""
- categoria_gasto: categoria ou ""
- entra_sai_gasto: "entrada" | "saida" | ""

# INTERPRETAÇÃO
- "gastos de hoje" → data = HOJE 00:00 a 23:59
- "quanto gastei esse mês" → dia 1 do mês até HOJE
- "gastos com mercado" → nome_gasto = "mercado"
- "o que recebi" → entra_sai_gasto = "entrada"
- "meus gastos de outubro" → outubro completo
